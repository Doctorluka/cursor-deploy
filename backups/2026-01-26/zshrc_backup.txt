proxy-on() {
    # 检查代理是否已启用
    if [ -n "$http_proxy" ] || [ -n "$https_proxy" ]; then
        echo "✅ 代理已经启用"
        echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"

    fi

    # 启用代理
    export http_proxy=http://127.0.0.1:7899

    echo "✅ 代理已启用"
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
